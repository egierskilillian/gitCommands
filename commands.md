+ clone #creates a copy of a repository
+ checkout #switches to another branch
  + -b #creates a new branch and switches to it
+ status #shows the state of the working directory
+ push #uploads local commits 
  + -u #sets the upstream branch 
+ pull #merges changes from the repository into current branch
+ add #adds changes to staging area
+ commit #records changes in repository
  + -m #adds a commit message
  + -a #stages all modified tracked files
  + -am #stages all tracked changes
+ branch #lists all local branches
  + -a #lists all local and remote branches
  + -d #deletes a branch locally
+ merge #merges a branch into the current branch
